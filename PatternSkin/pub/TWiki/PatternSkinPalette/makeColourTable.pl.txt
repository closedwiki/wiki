#! /usr/bin/perl -w
use strict;

my @colours;

while ( <> )
{
    chomp;
    push @colours, $_;
}

print qq{<table width="300px">\n};

foreach my $colour ( @colours )
{
    print "<tr>\n";
    print qq{<td style="background:#$colour" >};
        print $colour;    
    print "</td>\n";
    print "</tr>\n";
}

print "</table>\n";
